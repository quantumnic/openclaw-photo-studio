The WGSL shader for exposure adjustment:

Create crates/ocps-core/src/gpu/shaders/exposure.wgsl:
```wgsl
@group(0) @binding(0) var<storage, read> input: array<u32>;
@group(0) @binding(1) var<storage, read_write> output: array<u32>;
@group(0) @binding(2) var<uniform> params: ExposureParams;

struct ExposureParams {
    exposure_ev: f32,
    width: u32,
    height: u32,
    _pad: u32,
}

@compute @workgroup_size(64)
fn main(@builtin(global_invocation_id) id: vec3<u32>) {
    let idx = id.x;
    if idx >= params.width * params.height { return; }
    
    // Unpack RGB from u32 (packed as 0xRRGGBB00 in u16 pairs)
    // For simplicity: each u32 stores one channel value as u16
    let val = input[idx];
    let linear = f32(val) / 65535.0;
    
    // Apply exposure: multiply by 2^ev
    let factor = pow(2.0, params.exposure_ev);
    let adjusted = clamp(linear * factor, 0.0, 1.0);
    
    output[idx] = u32(adjusted * 65535.0);
}
```

pub trait TetherProvider: Send + Sync {
    fn name(&self) -> &str;
    fn discover_cameras(&self) -> Vec<TetheredCamera>;
    fn connect(&mut self, camera_id: &str) -> Result<(), TetherError>;
    fn disconnect(&mut self) -> Result<(), TetherError>;
    fn capture(&mut self) -> Result<Vec<u8>, TetherError>;  // returns raw file bytes
    fn live_view_frame(&mut self) -> Result<Option<Vec<u8>>, TetherError>;  // JPEG frame
}

pub struct TetheredCamera {
    pub id: String,
    pub name: String,          // "Canon EOS R5"
    pub provider: String,      // "gphoto2", "ptp", "sdk"
    pub connected: bool,
}

pub struct TetherSession {
    pub camera: TetheredCamera,
    pub import_folder: PathBuf,
    pub auto_import: bool,
    pub shot_count: u32,
}

// Mock provider for testing:
pub struct MockTetherProvider {
    connected: bool,
    shot_count: u32,
}

impl TetherProvider for MockTetherProvider {
    fn name(&self) -> &str { "Mock Camera" }
    fn discover_cameras(&self) -> Vec<TetheredCamera> {
        vec![TetheredCamera {
            id: "mock-001".into(),
            name: "Mock Camera (Test)".into(),
            provider: "mock".into(),
            connected: self.connected,
        }]
    }
    fn connect(&mut self, _id: &str) -> Result<(), TetherError> {
        self.connected = true; Ok(())
    }
    fn disconnect(&mut self) -> Result<(), TetherError> {
        self.connected = false; Ok(())
    }
    fn capture(&mut self) -> Result<Vec<u8>, TetherError> {
        self.shot_count += 1;
        // Return a minimal valid JPEG
        Ok(vec![0xFF, 0xD8, 0xFF, 0xE0]) // JPEG header stub
    }
    fn live_view_frame(&mut self) -> Result<Option<Vec<u8>>, TetherError> {
        Ok(None) // No live view in mock
    }
}
