    // Required:

    // export: get_parameters(out_ptr: i32) -> i32  (JSON parameter schema)
    // export: on_photo_selected(photo_id_ptr: i32, photo_id_len: i32)  (for UI plugins)

        // In production: download from download_url, verify sha256, save
        let plugin = self.plugins.iter().find(|p| p.id == plugin_id)

        // Create stub plugin directory
        let plugin_dir = target_dir.join(&plugin.id);
        std::fs::create_dir_all(&plugin_dir)?;
        
        // Write plugin.toml manifest
        std::fs::write(plugin_dir.join("plugin.toml"), format!(
            "[plugin]\nname = \"{}\"\nversion = \"{}\"\napi_version = \"1\"\ntype = \"{}\"\ndescription = \"{}\"\nauthor = \"{}\"\nentry_point = \"plugin.wasm\"\n\n[permissions]\nread_image = true\nwrite_image = true\nnetwork = false\n",
            plugin.name, plugin.version, plugin.plugin_type, plugin.description, plugin.author
        ))?;
        
        // Write stub WASM (empty but valid)
        let stub_wat = "(module (func (export \"plugin_init\") (result i32) (i32.const 0)))";
        let wasm_bytes = wat::parse_str(stub_wat).map_err(|e| MarketplaceError::InvalidPlugin(e.to_string()))?;
        std::fs::write(plugin_dir.join("plugin.wasm"), wasm_bytes)?;
        
        Ok(plugin_dir)
    }
}

### PART C: HDR Merge

Implement HDR merge as a built-in feature (not plugin).

Create crates/ocps-core/src/hdr/mod.rs:

pub struct HdrMergeSettings {
    pub deghosting: f32,        // 0.0 (none) to 1.0 (aggressive)
    pub auto_align: bool,
    pub base_exposure_index: usize,  // which input to use as base
}

pub struct HdrResult {
    pub image: RgbImage16,
    pub ev_range: f32,          // total exposure range (e.g., 6.0 EV)
    pub ghost_map: Option<Vec<f32>>,  // per-pixel ghosting probability
}

pub fn merge_hdr(
    exposures: &[(RgbImage16, f32)],  // (image, ev_offset)
    settings: &HdrMergeSettings,
) -> Result<HdrResult, HdrError>

Implementation approach (Mertens exposure fusion — simpler than true HDR):
1. For each input image, compute: contrast, saturation, well-exposedness weights
   - Contrast: Laplacian magnitude at each pixel
   - Saturation: standard deviation of RGB channels
   - Well-exposedness: Gaussian centered at 0.5 (mid-tone preferred)
2. Compute per-level Gaussian pyramids for each image
3. Compute Laplacian pyramids for each image
4. Blend pyramids using normalized weight pyramids
5. Reconstruct final image from blended Laplacian pyramid

For auto-align: use a simplified approach
   - Compute rough alignment via phase correlation on luminance channel
   - Apply integer pixel offset (subpixel alignment is complex, skip for now)

### PART D: Panorama Merge (Foundation)

Implement the foundation for panorama stitching.
Full stitching is complex — implement the preprocessing and a basic projection.

Create crates/ocps-core/src/panorama/mod.rs:

pub struct PanoramaSettings {
    pub projection: ProjectionType,
    pub overlap_percent: f32,   // expected overlap between shots
    pub blend_width: u32,       // pixels for seam blending
}

pub enum ProjectionType {
    Cylindrical,
    Spherical,
    Perspective,
}

pub struct PanoramaResult {
    pub image: RgbImage16,
    pub stitch_map: Vec<(usize, f32, f32)>,  // (source_index, x_offset, y_offset)
}

pub fn stitch_panorama(
    images: &[RgbImage16],
    settings: &PanoramaSettings,
) -> Result<PanoramaResult, PanoramaError>

Simple implementation:
1. For each pair of adjacent images: find offset using phase correlation
2. Lay images side by side with computed offsets
3. Alpha blend at seams (linear blend over blend_width pixels)
4. Return combined image
