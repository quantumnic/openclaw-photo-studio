    pub keywords: Vec<String>,
    pub date_filter: Option<DateFilter>,
    pub rating_filter: Option<u8>,
    pub camera_filter: Option<String>,
    pub location_filter: Option<String>,
    pub flag_filter: Option<String>,
}

pub fn parse_natural_query(query: &str) -> ParsedQuery {
    // Simple NLP-free parsing using heuristics:
    // "from 2025" / "in 2025" -> DateFilter::Year(2025)
    // "this year" -> DateFilter::ThisYear
    // "last month" -> DateFilter::LastMonth
    // "rated 4+" / "4 stars" -> rating_filter = 4
    // "picks" / "flagged" -> flag_filter = "pick"
    // "sony" / "nikon" / "canon" -> camera_filter
    // "paris" / "london" -> location_filter (search gps_city)
    // everything else -> keywords (search FTS)
    
    let mut parsed = ParsedQuery::default();
    let lower = query.to_lowercase();
    
    // Date patterns
    if let Some(year) = extract_year(&lower) {
        parsed.date_filter = Some(DateFilter::Year(year));
    }
    if lower.contains("this year") {
        parsed.date_filter = Some(DateFilter::ThisYear);
    }
    if lower.contains("last month") {
        parsed.date_filter = Some(DateFilter::LastMonth);
    }
    if lower.contains("today") {
        parsed.date_filter = Some(DateFilter::Today);
    }
    
    // Rating patterns
    if let Some(r) = extract_rating(&lower) {
        parsed.rating_filter = Some(r);
    }
    if lower.contains("picks") || lower.contains("flagged") {
        parsed.flag_filter = Some("pick".to_string());
    }
    if lower.contains("rejected") {
        parsed.flag_filter = Some("reject".to_string());
    }
    
    // Camera patterns
    for brand in &["sony", "nikon", "canon", "fuji", "olympus", "panasonic", "leica"] {
        if lower.contains(brand) {
            parsed.camera_filter = Some(brand.to_string());
            break;
        }
    }
    
    // Remaining words -> FTS keywords
    parsed.keywords = extract_keywords(&lower);
    
    parsed
}

export function ShortcutEditor() {
  const [bindings, setBindings] = createStore(loadBindings());
  const [editingAction, setEditingAction] = createSignal<string | null>(null);
  const [conflicts, setConflicts] = createSignal<Map<string, string[]>>(new Map());
  
  function startEdit(action: string) {
    setEditingAction(action);
    // Listen for next keydown event
    const handler = (e: KeyboardEvent) => {
      e.preventDefault();
      const key = formatKeyCombo(e);
      // Check for conflicts
      const conflict = findConflict(key, action, bindings);
      if (conflict) {
        // Show conflict dialog
      }
      updateBinding(action, key);
      setEditingAction(null);
      document.removeEventListener("keydown", handler);
    };
    document.addEventListener("keydown", handler);
  }
  
  function formatKeyCombo(e: KeyboardEvent): string {
    const parts = [];
    if (e.metaKey || e.ctrlKey) parts.push("cmd");
    if (e.shiftKey) parts.push("shift");
    if (e.altKey) parts.push("alt");
    if (e.key !== "Meta" && e.key !== "Control" && e.key !== "Shift" && e.key !== "Alt") {
      parts.push(e.key.toLowerCase());
    }
    return parts.join("+");
  }

}

pub struct PerfTracker {

pub struct PerfEvent {
    pub label: String,
    pub duration_ms: u64,
    pub timestamp: std::time::SystemTime,
}

impl PerfTracker {
    pub fn record<F, R>(label: &str, f: F) -> (R, u64)
    where F: FnOnce() -> R
    {
        let start = std::time::Instant::now();
        let result = f();
        let elapsed = start.elapsed().as_millis() as u64;
        (result, elapsed)
    }
}
