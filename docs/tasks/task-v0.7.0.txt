### PART D: Performance — Parallel Import

The import pipeline is currently sequential.
Add parallel processing for large imports.

In crates/ocps-catalog/src/db.rs, update import_folder:

```rust
use rayon::prelude::*;

pub fn import_folder_parallel(
    &mut self,
    folder_path: &Path,
    mode: ImportMode,
) -> Result<ImportResult, CatalogError> {
    // 1. Walk directory, collect file paths
    let files: Vec<PathBuf> = walkdir::WalkDir::new(folder_path)
        .into_iter()
        .filter_map(|e| e.ok())
        .filter(|e| is_supported_extension(e.path()))
        .map(|e| e.path().to_owned())
        .collect();
    
    // 2. Extract EXIF in parallel using rayon
    let metadata: Vec<(PathBuf, Option<FileMetadata>)> = files
        .par_iter()
        .map(|path| {
            let meta = extract_file_metadata(path).ok();
            (path.clone(), meta)
        })
        .collect();
    
    // 3. Insert into DB (sequential - SQLite doesn't handle parallel writes well)
    let mut result = ImportResult::default();
    for (path, meta) in metadata {
        match self.insert_photo(&path, meta) {
            Ok(inserted) => if inserted { result.inserted += 1 } else { result.skipped += 1 },
            Err(e) => result.errors.push(e.to_string()),
        }
        result.total += 1;
    }
    
    Ok(result)
}
```
